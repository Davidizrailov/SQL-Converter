using System;

class Program
{
    static void Main()
    {
        // Example SQL query to be converted
        string sqlQuery = "SELECT TOP 5 Name, Age FROM Employees WHERE Department = 'HR' ORDER BY Age DESC;";

        // Simulate some processing
        Console.WriteLine("Original SQL Query:");
        Console.WriteLine(sqlQuery);

        // Manipulate the query (for demonstration purposes)
        string modifiedQuery = ModifyQuery(sqlQuery);

        // Output the modified query
        Console.WriteLine("\nModified SQL Query:");
        Console.WriteLine(modifiedQuery);
    }

    static string ModifyQuery(string query)
    {
        // Simple manipulation: Replace SELECT with a comment for demonstration
        return query.Replace("SELECT", "-- SELECT");
    }
}
